45 @staticmethod
46 def compute_band_energies(signal: np.ndarray, num_bands: int = 32) -> np.ndarray:
47 """
48        Compute energy in frequency bands.
49
50        Like a spectrum analyzer: Split signal into bands, measure energy per band.
51        """
52        band_size = len(signal) // num_bands
53        energies = np.zeros(num_bands)
54
55 for i in range(num_bands):
56            start = i * band_size
57            end = start + band_size if i < num_bands - 1 else len(signal)
58            band = signal[start:end]
59            energies[i] = np.sum(band ** 2) # Energy = sum of squares
60
61 return energies
62
63 @staticmethod
64 def compute_interference_energy(sig_a: np.ndarray, sig_b: np.ndarray) -> float:
65 """
66        Compute interference energy when two signals are combined.
67
68        E(a+b) = E(a) + E(b) + 2*dot(a,b)
69
70        The interference term 2*dot(a,b) tells us:
71        - Positive: Constructive interference (signals reinforce)
72        - Negative: Destructive interference (signals cancel)
73        """
74 return 2 * np.dot(sig_a, sig_b)
75
76 @staticmethod
77 def compute_resonance_profile(sig_a: np.ndarray, sig_b: np.ndarray, num_bands: int = 32) -> np.ndarray:
78 """
79        Compute per-band resonance.
80
81        Where do the signals resonate (reinforce each other)?
82        Where do they cancel?
83        """
84        band_size = len(sig_a) // num_bands
85        resonance = np.zeros(num_bands)
86
87 for i in range(num_bands):
88            start = i * band_size
89            end = start + band_size if i < num_bands - 1 else len(sig_a)
90
91            band_a = sig_a[start:end]
92            band_b = sig_b[start:end]
93
94 # Resonance = correlation in this band
95            resonance[i] = np.dot(band_a, band_b)
96
97 return resonance
98
99 @staticmethod
100 def spectral_centroid(energies: np.ndarray) -> float:
101 """
102        Spectral centroid: "Center of mass" of the spectrum.
103
104        Bright sounds have high centroid, dark sounds have low centroid.
105        """
106        total = np.sum(energies)
107 if total < 1e-10:
108 return len(energies) / 2
109
110        frequencies = np.arange(len(energies))
111 return np.sum(frequencies * energies) / total
112
113 @staticmethod
114 def spectral_spread(energies: np.ndarray, centroid: float = None) -> float:
115 """
116        Spectral spread: How "wide" is the spectrum?
117
118        Narrow = focused on specific frequencies
119        Wide = broadband noise
120        """
121 if centroid is None:
122            centroid = AudioToolkit.spectral_centroid(energies)
123
124        total = np.sum(energies)
125 if total < 1e-10:
126 return 0
127
128        frequencies = np.arange(len(energies))
129        variance = np.sum(((frequencies - centroid) ** 2) * energies) / total
130 return np.sqrt(variance)
131
132 @staticmethod
133 def spectral_flatness(energies: np.ndarray) -> float:
134 """
135        Spectral flatness: Tonality vs Noise.
136
137        1.0 = White noise (flat spectrum)
138        0.0 = Pure tone (single peak)
139
140        Geometric mean / Arithmetic mean
141        """
142        energies = np.maximum(energies, 1e-10) # Avoid log(0)
143
144        geometric_mean = np.exp(np.mean(np.log(energies)))
145        arithmetic_mean = np.mean(energies)
146
147 return geometric_mean / arithmetic_mean

164                 num_bands: int = 32,
165                 energy_threshold_percentile: float = 75,
166                 use_phase: bool = True):
167 """
168        Args:
169            num_bands: Number of frequency bands
170            energy_threshold_percentile: Bands above this percentile are "active"
171            use_phase: Also consider sign (phase) of dominant dimension per band
172        """
173        self.num_bands = num_bands
174        self.energy_threshold_percentile = energy_threshold_percentile
175        self.use_phase = use_phase
176
177 # Storage
178        self.doc_ids = []
179        self.doc_embeddings = None
180        self.doc_band_energies = None

187 def _compute_signature(self, embedding: np.ndarray) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
188 """
189        Compute band signature for an embedding.
190
191        Returns:
192            energies: Energy per band
193            phases: Dominant phase per band (+1 or -1)
194            active: Boolean mask of active bands
195        """
196        band_size = len(embedding) // self.num_bands
197        energies = np.zeros(self.num_bands)
198        phases = np.zeros(self.num_bands)
199
200 for i in range(self.num_bands):
201            start = i * band_size
202            end = start + band_size if i < self.num_bands - 1 else len(embedding)
203            band = embedding[start:end]
204
205 # Energy
206            energies[i] = np.sum(band ** 2)
207
208 # Phase: Sign of the maximum absolute value
209            max_idx = np.argmax(np.abs(band))
210            phases[i] = 1 if band[max_idx] > 0 else -1
211
212 # Active bands: Above threshold
213        threshold = np.percentile(energies, self.energy_threshold_percentile)
214        active = energies > threshold
215
216 return energies, phases, active

239                    key = (int(band_idx), int(phases[band_idx]))

245               candidate_multiplier: int = 20) -> List[Tuple[str, float]]:
246 """
247        Resonance-based search.
248
249        1. Find documents with overlapping active bands (resonance candidates)
250        2. Compute interference energy for fine ranking
251        """

253

255        votes = defaultdict(int)
256
257 for band_idx in np.where(query_active)[0]:
258 if self.use_phase:
259                key = (int(band_idx), int(query_phases[band_idx]))
260 else:
261                key = int(band_idx)
262
263 for doc_idx in self.band_index.get(key, []):
264                votes[doc_idx] += 1
265
266 # Get top candidates
267        num_candidates = min(top_k * candidate_multiplier, len(self.doc_ids))
268
269 if votes:
270            candidate_indices = sorted(votes.keys(), key=lambda x: -votes[x])[:num_candidates]
271 else:
272 # Fallback: random candidates
273            candidate_indices = list(range(min(num_candidates, len(self.doc_ids))))
274
275 # Stage 2: Compute interference energy for fine ranking
276        query_norm = query_embedding / (np.linalg.norm(query_embedding) + 1e-10)
277
278        results = []
279

281            doc_emb = self.doc_embeddings[doc_idx]

283
284 # Interference energy (= 2 * cosine for normalized vectors)
285            interference = AudioToolkit.compute_interference_energy(query_norm, doc_norm)
286
287            results.append((self.doc_ids[doc_idx], -interference))
288
289        results.sort(key=lambda x: x[1])

297class HarmonicResonanceIndex:
298 """
299    Use harmonic relationships for indexing.
300
301    In music: Notes that are harmonically related sound "good together".
302    E.g., C and G (perfect fifth) resonate.
303
304    For embeddings: Dimensions that are "harmonically related" 
305    (e.g., d and 2d, d and 3d) should be considered together.
306    """
307
308 def __init__(self,
309                 num_fundamentals: int = 64,
310                 harmonics: List[int] = [2, 3, 4, 5],
311                 top_k_fundamentals: int = 8):
312 """
313        Args:
314            num_fundamentals: Consider first N dimensions as potential fundamentals
315            harmonics: Which harmonic ratios to check
316            top_k_fundamentals: How many fundamentals to use for signature
317        """
318        self.num_fundamentals = num_fundamentals
319        self.harmonics = harmonics
320        self.top_k_fundamentals = top_k_fundamentals
321
322        self.doc_ids = []
323        self.doc_embeddings = None
324        self.doc_harmonic_signatures = {}
325
326 # Inverted index: (fundamental, harmonic_pattern) → [doc_indices]
327        self.harmonic_index = defaultdict(list)
328
329 def _compute_harmonic_signature(self, embedding: np.ndarray) -> List[Tuple[int, Tuple]]:
330 """
331        Find dominant fundamentals and their harmonic patterns.
332
333        Returns: List of (fundamental_dim, harmonic_pattern)
334        """
335        dim = len(embedding)
336
337 # Score each potential fundamental by harmonic coherence
338        fundamental_scores = []
339
340 for f in range(1, self.num_fundamentals + 1):
341 # Fundamental value
342            f_val = embedding[f - 1]
343
344 # Check harmonics
345            harmonic_pattern = []
346            coherence = abs(f_val)
347
348 for h in self.harmonics:
349                h_idx = f * h - 1
350 if h_idx < dim:
351                    h_val = embedding[h_idx]
352 # Coherent if same sign and significant
353 if f_val * h_val > 0: # Same sign
354                        coherence += abs(h_val) / h  # Weight by harmonic number
355                        harmonic_pattern.append(1)
356 else:
357                        harmonic_pattern.append(0)
358 else:
359                    harmonic_pattern.append(-1) # Out of range
360
361            fundamental_scores.append((f, coherence, tuple(harmonic_pattern)))
362
363 # Get top fundamentals
364        fundamental_scores.sort(key=lambda x: -x[1])
365
366        signature = []
367 for f, score, pattern in fundamental_scores[:self.top_k_fundamentals]:
368            signature.append((f, pattern))
369
370 return signature

372 def build(self, doc_ids: List[str], embeddings: np.ndarray):

374        self.doc_ids = doc_ids
375        self.doc_embeddings = embeddings
376

378            signature = self._compute_harmonic_signature(emb)
379            self.doc_harmonic_signatures[doc_ids[i]] = signature
380
381 for fundamental, pattern in signature:
382                key = (fundamental, pattern)
383                self.harmonic_index[key].append(i)
384
385 def search(self, query_embedding: np.ndarray, top_k: int = 10,
386               candidate_multiplier: int = 20) -> List[Tuple[str, float]]:
387 """Search by harmonic pattern matching."""
388        query_signature = self._compute_harmonic_signature(query_embedding)
389
390 # Vote counting
391        votes = defaultdict(int)
392
393 for fundamental, pattern in query_signature:
394            key = (fundamental, pattern)
395 for doc_idx in self.harmonic_index.get(key, []):
396                votes[doc_idx] += 1
397
398 # Get candidates
399        num_candidates = min(top_k * candidate_multiplier, len(self.doc_ids))
400
401 if votes:
402            candidate_indices = sorted(votes.keys(), key=lambda x: -votes[x])[:num_candidates]
403 else:
404            candidate_indices = list(range(min(num_candidates, len(self.doc_ids))))
405
406 # Fine ranking with cosine
407        query_norm = query_embedding / (np.linalg.norm(query_embedding) + 1e-10)
408
409        results = []
410 for doc_idx in candidate_indices:
411            doc_emb = self.doc_embeddings[doc_idx]
412            doc_norm = doc_emb / (np.linalg.norm(doc_emb) + 1e-10)
413            sim = np.dot(query_norm, doc_norm)
414            results.append((self.doc_ids[doc_idx], -sim))
415
416        results.sort(key=lambda x: x[1])
417 return results[:top_k]

424class SpectralCentroidIndex:
425 """
426    Index by spectral characteristics (centroid, spread, flatness).
427
428    Intuition: 
429    - "Bright" embeddings have high centroid (energy in high dimensions)
430    - "Dark" embeddings have low centroid (energy in low dimensions)
431
432    Similar semantic content might have similar "brightness".
433    """
434
435 def __init__(self,
436                 num_bands: int = 32,
437                 centroid_buckets: int = 16,
438                 spread_buckets: int = 8):
439 """
440        Args:
441            num_bands: For band energy computation
442            centroid_buckets: Quantization levels for centroid
443            spread_buckets: Quantization levels for spread
444        """
445        self.num_bands = num_bands
446        self.centroid_buckets = centroid_buckets
447        self.spread_buckets = spread_buckets
448
449        self.doc_ids = []
450        self.doc_embeddings = None
451        self.doc_features = {} # doc_id → (centroid_bucket, spread_bucket, flatness)
452
453 # Inverted index
454        self.centroid_index = defaultdict(list) # centroid_bucket → [doc_indices]
455
456 def _compute_spectral_features(self, embedding: np.ndarray) -> Tuple[float, float, float]:
457 """Compute spectral features."""
458        energies = AudioToolkit.compute_band_energies(embedding, self.num_bands)
459
460        centroid = AudioToolkit.spectral_centroid(energies)
461        spread = AudioToolkit.spectral_spread(energies, centroid)
462        flatness = AudioToolkit.spectral_flatness(energies)
463
464 return centroid, spread, flatness
465
466 def build(self, doc_ids: List[str], embeddings: np.ndarray):
467 """Build the spectral index."""
468        self.doc_ids = doc_ids
469        self.doc_embeddings = embeddings
470
471 # First pass: compute all features to find ranges
472        all_centroids = []
473        all_spreads = []
474
475 for emb in embeddings:
476            c, s, f = self._compute_spectral_features(emb)
477            all_centroids.append(c)
478            all_spreads.append(s)
479
480        self.centroid_min = min(all_centroids)
481        self.centroid_max = max(all_centroids)
482        self.spread_min = min(all_spreads)
483        self.spread_max = max(all_spreads)
484
485 # Second pass: index
486 for i, emb in enumerate(embeddings):
487            c, s, f = self._compute_spectral_features(emb)
488
489 # Quantize
490            c_bucket = int((c - self.centroid_min) / (self.centroid_max - self.centroid_min + 1e-10) * self.centroid_buckets)
491            c_bucket = min(c_bucket, self.centroid_buckets - 1)
492
493            s_bucket = int((s - self.spread_min) / (self.spread_max - self.spread_min + 1e-10) * self.spread_buckets)
494            s_bucket = min(s_bucket, self.spread_buckets - 1)
495
496            self.doc_features[doc_ids[i]] = (c_bucket, s_bucket, f)
497
498 # Index by centroid (and neighboring buckets)
499 for offset in [-1, 0, 1]:
500                bucket = c_bucket + offset
501 if 0 <= bucket < self.centroid_buckets:
502                    self.centroid_index[bucket].append(i)
503
504 def search(self, query_embedding: np.ndarray, top_k: int = 10,
505               candidate_multiplier: int = 20) -> List[Tuple[str, float]]:
506 """Search by spectral similarity."""
507        c, s, f = self._compute_spectral_features(query_embedding)
508
509 # Quantize query
510        c_bucket = int((c - self.centroid_min) / (self.centroid_max - self.centroid_min + 1e-10) * self.centroid_buckets)
511        c_bucket = min(max(c_bucket, 0), self.centroid_buckets - 1)
512
513 # Get candidates from same and neighboring centroid buckets
514        candidates = set()
515 for offset in [-2, -1, 0, 1, 2]:
516            bucket = c_bucket + offset
517 if 0 <= bucket < self.centroid_buckets:
518                candidates.update(self.centroid_index.get(bucket, []))
519
520        num_candidates = min(top_k * candidate_multiplier, len(self.doc_ids))
521        candidate_list = list(candidates)[:num_candidates]
522
523 if len(candidate_list) < num_candidates:
524 # Add random docs if not enough
525            remaining = set(range(len(self.doc_ids))) - candidates
526            candidate_list.extend(list(remaining)[:num_candidates - len(candidate_list)])
527
528 # Fine ranking
529        query_norm = query_embedding / (np.linalg.norm(query_embedding) + 1e-10)
530
531        results = []
532 for doc_idx in candidate_list:
533            doc_emb = self.doc_embeddings[doc_idx]
534            doc_norm = doc_emb / (np.linalg.norm(doc_emb) + 1e-10)
535            sim = np.dot(query_norm, doc_norm)
536            results.append((self.doc_ids[doc_idx], -sim))
537
538        results.sort(key=lambda x: x[1])
539 return results[:top_k]

546class ConstructiveInterferenceSearch:
547 """
548    Direct interference-based search.
549
550    The idea: For each query, find documents where combining them
551    produces MAXIMUM constructive interference.
552
553    This is mathematically equivalent to cosine similarity,
554    but the framing helps us think about it differently.
555    """
556
557 def __init__(self,
558                 num_bands: int = 32,
559                 use_band_weighting: bool = True):
560 """
561        Args:
562            num_bands: Number of frequency bands for analysis
563            use_band_weighting: Weight bands by query energy
564        """
565        self.num_bands = num_bands
566        self.use_band_weighting = use_band_weighting
567
568        self.doc_ids = []
569        self.doc_embeddings = None
570        self.doc_band_embeddings = None # Pre-split into bands
571
572 def build(self, doc_ids: List[str], embeddings: np.ndarray):
573 """Build index."""
574        self.doc_ids = doc_ids
575        self.doc_embeddings = embeddings
576
577 # Pre-compute band splits for faster search
578        n_docs, dim = embeddings.shape
579        band_size = dim // self.num_bands
580
581        self.doc_band_embeddings = embeddings.reshape(n_docs, self.num_bands, band_size)
582
583 def search(self, query_embedding: np.ndarray, top_k: int = 10) -> List[Tuple[str, float]]:
584 """
585        Search by constructive interference.
586
587        For each band, compute interference and weight by query band energy.
588        """
589        dim = len(query_embedding)
590        band_size = dim // self.num_bands
591
592        query_bands = query_embedding.reshape(self.num_bands, band_size)
593
594 # Compute query band energies for weighting
595 if self.use_band_weighting:
596            query_band_energies = np.sum(query_bands ** 2, axis=1)
597            query_band_energies = query_band_energies / (np.sum(query_band_energies) + 1e-10)
598 else:
599            query_band_energies = np.ones(self.num_bands) / self.num_bands
600
601 # Compute interference for all docs
602 # interference[d, b] = 2 * dot(query_band[b], doc_band[d, b])
603
604 # Shape: (n_docs, num_bands)
605        band_interferences = 2 * np.einsum('nb,dnb->dn', query_bands, self.doc_band_embeddings)
606
607 # Weight by query band energy
608        weighted_interference = band_interferences @ query_band_energies
609
610 # Get top-k
611        top_indices = np.argsort(weighted_interference)[::-1][:top_k]
612
613        results = []
614 for idx in top_indices:
615            results.append((self.doc_ids[idx], -weighted_interference[idx]))
616
617 return results

624class AudioSemanticIndex:
625 """
626    Combines multiple audio-inspired approaches.
627
628    Uses a voting system where different indices propose candidates,
629    then final ranking is done by interference energy.
630    """
631
632 def __init__(self,
633                 num_bands: int = 32,
634                 use_resonance: bool = True,
635                 use_harmonic: bool = True,
636                 use_spectral: bool = True):
637
638        self.num_bands = num_bands
639
640        self.indices = {}
641
642 if use_resonance:
643            self.indices["resonance"] = ResonanceIndex(num_bands=num_bands)
644 if use_harmonic:
645            self.indices["harmonic"] = HarmonicResonanceIndex()
646 if use_spectral:
647            self.indices["spectral"] = SpectralCentroidIndex(num_bands=num_bands)
648
649        self.doc_ids = []
650        self.doc_embeddings = None
651
652 def build(self, doc_ids: List[str], embeddings: np.ndarray):
653 """Build all indices."""
654        self.doc_ids = doc_ids
655        self.doc_embeddings = embeddings
656
657 for name, index in self.indices.items():
658            index.build(doc_ids, embeddings)
659
660 def search(self, query_embedding: np.ndarray, top_k: int = 10,
661               candidates_per_index: int = 100) -> List[Tuple[str, float]]:
662 """
663        Search using all indices and combine results.
664        """
665 # Gather candidates from all indices
666        all_candidates = set()
667
668 for name, index in self.indices.items():
669            results = index.search(query_embedding, top_k=candidates_per_index)
670 for doc_id, _ in results:
671                doc_idx = self.doc_ids.index(doc_id)
672                all_candidates.add(doc_idx)
673
674 # Fine ranking with interference
675        query_norm = query_embedding / (np.linalg.norm(query_embedding) + 1e-10)
676
677        results = []
678 for doc_idx in all_candidates:
679            doc_emb = self.doc_embeddings[doc_idx]
680            doc_norm = doc_emb / (np.linalg.norm(doc_emb) + 1e-10)
681
682            interference = AudioToolkit.compute_interference_energy(query_norm, doc_norm)
683            results.append((self.doc_ids[doc_idx], -interference))
684
685        results.sort(key=lambda x: x[1])
686 return results[:top_k]
